# parameter-process-mapping: list of lists
#   - each line is one process
#   - number of lists per line is the number of process options
#   - for each process option the parameters active in this
#     process must be listed
#   - empty lists are possible (when process option does not
#     depend on parameters)
#   - parameter names must correspond to names in 'parameters.txt')
# 
[[par_x01], [par_x01, par_x02]]                  # process 1: option [A1, A2]
[[par_x02], [par_x03], [par_x04, par_x05]]       # process 2: option [B1, B2, B3]
[[par_x06], [par_x03, par_x07]]                  # process 3: option [C1, C2]
